	\section{Architecture and Compliance}
	\begin{frame}
		\frametitle{Technical Requirements}
		Reference: \href{https://people.clas.ufl.edu/jnowell/latex-accessibility/}{Dr. Jason Nowell}.
		\vspace{0.5cm}
		
		\textbf{Critical Workflow Components:}
		\begin{itemize}
			\item \textbf{Engine:} Use \texttt{LuaLaTeX} for its superior handling of OpenType math fonts, which is critical for MathML generation in order to populate math tags.
			\item \textbf{Kernel:} Use the Phase 3 Beta kernel (released Nov 1st, 2025, or later).
			\item \textbf{Preamble:} Use preamble and accessibility packages provided in this presentation and reference.  For slides use (highly experimental) document class \href{https://ctan.math.illinois.edu/macros/latex/contrib/ltx-talk/ltx-talk.pdf}{\texttt{ltx-talk}}.
			\item \textbf{Validation:}
			\begin{itemize}
				\item \textbf{Structural:} Use \href{https://openpreservation.org/tools/verapdf/}{veraPDF} (Industry Standard).  Also recommend  checker \href{https://pac.pdf-accessibility.org/en}{PAC 2026} with detailed report and built-in screen reader.
				\item \textbf{Semantic:} Verify with screen readers (NVDA, JAWS, or VoiceOver).
			\end{itemize}
			\item \textbf{Best Practices:} Prioritize structural simplicity to ensure tag-tree integrity.
		\end{itemize}
	\end{frame}

	\section{Math/Verbatim}
	
	\begin{frame}
		\frametitle{Automatic MathML Support}
		An example of a slide containing math.
		\begin{itemize}
			\item<1-2> \textbf{Sampling distribution:} \\
			Let $T(Y_1, Y_2, \dots, Y_n)$ be some \textit{statistic}, i.e., some function of the data.
			Then, we call the probability distribution of $T(Y_1, Y_2, \dots, Y_n)$ the \emph{sampling distribution}.
			
			
			\item<2> \textbf{Standard error:} \\
			The \emph{standard error} of an estimator $\hat{\theta}$ is the standard deviation of the sampling distribution:
			\[ \mathrm{SE}\left(\hat{\theta}\right) = \sqrt{E\left[\left(\hat{\theta}-\theta\right)^2\right]} \]
		\end{itemize}
		\begin{block}<3>{Block}
			This is a block, like a theorem.  Background colors can cause issues, even if color scheme is appropriate.
		\end{block}
	\end{frame}

	\section{Toggle Visibility Logic}
	\begin{frame}
		\frametitle{Toggle Visibility Logic}
		
		Current Mode: \ifshowsolutions \textbf{Solutions Visible} \else \textbf{Student Handout} \fi
		
		In preamble comment out one:
		\begin{itemize}
			\item[] \textbackslash showsolutionstrue
			\item[] \textbackslash showsolutionsfalse
		\end{itemize}
		
		
		% The \sol command handles the conditional logic internally
		\sol[3cm]{
			\begin{align*}
				V\left(\hat\mu\right) &= V\left(\frac{1}{n}\sum_{i=1}^n Y_i\right)\\
				&= \frac{1}{n^2}\sum_{i=1}^n V(Y_i) = \frac{\sigma^2}{n}\\
				\Longrightarrow \mathrm{SE}\left(\hat\mu\right) &= \frac{\sigma}{\sqrt{n}}
			\end{align*}
		}
		
		\begin{block}{Preamble Command Used}
			\small
			\texttt{\textbackslash sol[space\_when\_false]\{content when true\}}\\ allows for a single-source workflow, generating an ``Instructor'' and ``Student'' version easily.
		\end{block}
	\end{frame}

	\begin{frame*}
		\frametitle{Temporal Overlays}
		\fade<1> With the use of \texttt{\textbackslash temporal}, can create faded overlays whereby a before, current, and after options can be chosen.
		\begin{verbatim}
			\NewDocumentCommand\fade{D<>{all}}{%
				\temporal<#1>{\color{black!70}}{\color{black}}{\color{black!10}}
			}
		\end{verbatim}
		\vspace{-0.5cm}
		
		\begin{enumerate}
			\fade<2> \item First item.
			\fade<3> \item Second item.
			\fade<4-5>\item Third item.
			\fade<6>\item Fourth item (while also displaying third).
		\end{enumerate}
		\vspace{1cm}
		
		\fade<7->{\textit{Note:} Unlike standard overlays, temporal overlays for lists retain visible bullets.  The use of \texttt{\textbackslash temporal} inside a list, the new kernel's tagging engine (tagpdf) forces the item bullet to be a protected "Label"}
		
	\end{frame*}
