% --- Comment out chunks ---

	% --- Core Accessibility & Structure ---
	% Sets the document language for screen readers (American English)
	\usepackage[american]{babel}
	% Configures strict PDF tagging parameters:
	% - math/alt/use: Enables MathML generation so screen readers can read equations
	% - activate: Turns on PDF structure tagging
	% - interwordspace=true: Helps screen readers separate words correctly
	% - uncompress: Keeps the PDF uncompressed, which is required for some accessibility checkers
	\tagpdfsetup{
		math/alt/use,
		activate,
		interwordspace=true,
		uncompress
	}  
	
	% --- Math Fonts and others ---
	% Standard packages for images, high-quality tables, and colors

	\usepackage{tikz}

	% \frenchspacing ensures single spacing after sentences, which is standard for digital reading
	\frenchspacing

	\newcommand{\vs}{\vspace{0.2cm}}
	% ======================================================================
	% FRAME DESIGN & COLOR CONFIGURATION
	% ======================================================================
	% 1. Style the container (The Banner Background)
	% Sets the default top header banner to UF Blue with white text
	\EditInstance{header}{std}{
		background-color = UFBlue,
		color = white,
		print-frame-title = true
	}  
	
	% 2. Style the text inside the container (The Typography)
	% Formats the slide frametitle text to be bold and huge
	\EditInstance{frametitle}{header}{
		font = \bfseries\huge
	}  
	
	% 3. Banner Visibility Toggles
	% a. Turn banner off (Make it transparent and heightless)
	% Useful for full-screen images, transition slides, or title pages
	\newcommand{\banneroff}{%
		\EditInstance{header}{std}{
			background-color = white, % Matches slide background
			color = white,            % Hides title text
			print-frame-title = false
		}
	}
	% b. Restore standard Blue Banner
	\newcommand{\banneron}{%
		\EditInstance{header}{std}{
			background-color = UFBlue,
			color = white,
			print-frame-title = true
		}
	}  
	
	% --- Footer: Frame Numbering (X/Y) ---
	\makeatletter
	% Define the content for our progress indicator (Current Slide / Total Slides)
	\newcommand{\@slideprogress}{\@framenumber/\@totalframes}
	% Define a truly empty command to act as a placeholder element on the left side
	\newcommand{\@emptycontent}{}
	\makeatother  
	
	% Register BOTH elements
	% Note: We leave the brackets empty {} because 'talk' elements
	% automatically look for their matching \@command name.
	\DeclareInstance{footer-element}{empty}{talk}{}
	\DeclareInstance{footer-element}{slideprogress}{talk}{}  
	
	% Configure the footer instance
	% Sets a white footer with black text, separating the empty placeholder and the slide numbers 
	% with an \hfill so that the page numbers are pushed flush to the right.
	\EditInstance{footer}{std}{
		background-color = white,
		color            = black,
		% Listing two elements forces the 'separator' (\hfill) to trigger between them
		element-order    = {empty, slideprogress},
		separator        = \hfill,
		left-hspace      = 5mm,
		right-hspace     = 5mm,
		font             = \small\sffamily
	}

	% ======================================================================
	% CUSTOM SEMANTIC BLOCKS (Corrected Left Border)
	% =====================================================================
	% Creates custom box environments (Theory, Example, Remark) that draw a thick 
	% colored vertical line on the left to visually separate them from standard text.
	\newenvironment{theory}[1][]{%
		\vspace{0.5em}\noindent
		\textcolor{UFBlue}{\vrule width 3pt}\hspace{0.5em}%
		\begin{minipage}[t]{\dimexpr\textwidth-3pt-0.5em\relax}
			\begin{block}{\strut\color{UFBlue}\bfseries Theory\if\relax\detokenize{#1}\relax\else: #1\fi}
				\vspace{2pt}}{%
				\vspace{2pt}\end{block}
		\end{minipage}\vspace{0.5em}}  
	
	\newenvironment{example}[1][]{%
		\vspace{0.5em}\noindent
		\textcolor{UFOrange}{\vrule width 3pt}\hspace{0.5em}%
		\begin{minipage}[t]{\dimexpr\textwidth-3pt-0.5em\relax}
			\begin{block}{\strut\color{UFOrange}\bfseries Example\if\relax\detokenize{#1}\relax\else: #1\fi}
				\vspace{2pt}}{%
				\vspace{2pt}\end{block}
		\end{minipage}\vspace{0.5em}}  
	
	\newenvironment{remark}[1][]{%
		\vspace{0.5em}\noindent
		\textcolor{UFGreen}{\vrule width 3pt}\hspace{0.5em}%
		\begin{minipage}[t]{\dimexpr\textwidth-3pt-0.5em\relax}
			\begin{block}{\strut\color{UFGreen}\bfseries Remark\if\relax\detokenize{#1}\relax\else: #1\fi}
				\vspace{2pt}}{%
				\vspace{2pt}\end{block}
		\end{minipage}\vspace{0.5em}}  
	
	\newenvironment{definition}[1][]{%
		\vspace{0.5em}\noindent
		\textcolor{UFRed}{\vrule width 3pt}\hspace{0.5em}%
		\begin{minipage}[t]{\dimexpr\textwidth-3pt-0.5em\relax}
			\begin{block}{\strut\color{UFRed}\bfseries Definition\if\relax\detokenize{#1}\relax\else: #1\fi}
				\vspace{2pt}}{%
				\vspace{2pt}\end{block}
		\end{minipage}\vspace{0.5em}}

	% 1. Remap frame titles to logically follow H1 sections
	% Ensures Canvas Ally doesn't flag a skipped heading level (fixing the jump from H1 -> H4)
	\tagpdfsetup{role / new-tag = frametitle / H2}  
	
	% 2. Optimized Section Transitions to satisfy Ally/Canvas Logical Structure
	% Wraps the standard \section command to automatically insert a hidden plain slide.
	% This prevents strict accessibility checkers from flagging H1 sections as "empty".
	\let\oldsection\section
	\renewcommand{\section}[1]{%
		\oldsection{#1}%
		\banneroff
		\begin{frame}[plain]
			\tableofcontents[currentsection]
			% This invisible text "fills" the H1 section so Ally doesn't see it as empty
			\phantom{Starting section #1}
		\end{frame}
		\banneron
	}
